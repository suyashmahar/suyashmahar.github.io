module fifo(clk,reset,rd,wr,w_data,r_data,empty,full);
input clk,reset,rd,wr;
input [0:7]w_data;
output reg empty,full;
output reg [0:7]r_data;
integer i,front,rear;
reg [0:7]stack[0:7];
initial begin
i=0;
stack[0]=-1;
stack[1]=-1;
stack[2]=-1;
stack[3]=-1;
stack[4]=-1;
stack[5]=-1;
stack[6]=-1;
stack[7]=-1;
front=-1;
rear=-1;
empty=1;
full=0;
end
always @(posedge clk)
begin
if(rd==1&&empty==0)
begin
if(front==rear)
begin
r_data=stack[front];
i=i-1;
full=0;
stack[front]=-1;
front=-1;
rear=-1;
end
else begin
r_data=stack[front];
i=i-1;
front=front+1;
end
if(front==7)
begin
r_data=stack[front];
i=i-1;
full=0;
stack[front]=-1;
front=0;
end
if(front<0)
begin
empty=1;
end
end
if(rd==1&&empty==0)
begin
if(front==7)
begin
r_data=stack[front];
i=i-1;
full=0;
stack[front]=-1;
front=0;
end
if(front==rear)
begin
r_data=stack[front];
i=i-1;
stack[front]=-1;
front=front+1;
end
if(front<0)
begin
empty=1;
end
end
if(wr==1&&full==0)
begin
if(front==-1)
begin
rear=0;
stack[rear]=w_data;
end
end
if(reset==1)
begin
i=0;
stack[0]=-1;
stack[1]=-1;
stack[2]=-1;
stack[3]=-1;
stack[4]=-1;
stack[5]=-1;
stack[6]=-1;
stack[7]=-1;
front=-1;
rear=-1;


empty=1;
full=0;
end
end
endmodule
