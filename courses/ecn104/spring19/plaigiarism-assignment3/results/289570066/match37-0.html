module fifo(
clock, reset, rd, wr, w_data,r_data, empty, full
);
input reset;
input rd;
input wr;
input clock;
input[7:0] w_data;
output full, empty;
output reg[7:0] r_data;
reg [2:0] count, r_head, w_head;


reg [7:0] queue [7:0];
assign full = (count==7) ? 1'b1:1'b0;
assign empty = (count==0) ? 1'b1:1'b0;
initial begin
count = 0;
r_head = 7;
w_head = 7;
end
always @ (posedge clock) begin
if (reset) begin
r_head = 7;
w_head = 7;
count = 0;
end
else if (wr) begin
if(!full) begin
queue[w_head] = w_data;
w_head = w_head - 1;
count = count + 1;
end
end
else if(rd) begin
if(!empty) begin
r_data = queue[r_head];
r_head = r_head - 1;
count = count - 1;
end
end
if(w_head == -1)
w_head = 7;
if(r_head == -1)
r_head = 7;
end
endmodule
