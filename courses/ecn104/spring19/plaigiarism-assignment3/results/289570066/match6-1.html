module fifo(ghadi,reset,rd,wr,w_data,r_data,emptyqueue,fullqueue);
input reset,rd,wr,ghadi;
input [8-1:0] w_data;
output fullqueue,emptyqueue;
output reg [8-1:0] r_data;
reg[3-1:0] count , r_head, w_head ;
reg [8-1:0] queue [8-1:0] ;
assign fullqueue = (count==7)? 1'b1:1'b0;
assign emptyqueue = (count==0)? 1'b1:1'b0;
initial begin
count = 0;
r_head = 7;
w_head = 7 ;
end
always @ (posedge ghadi) begin
if (reset) begin
r_head = 7;
w_head = 7;
count = 0;
end
else if(wr) begin
if(!fullqueue) begin
queue[w_head] = w_data;
w_head = w_head - 1;
count = count+1;
end
end
else if(rd) begin
if(!emptyqueue) begin
r_data = queue[r_head];
r_head = r_head - 1;
count = count - 1;
end
end

if(r_head == -1) r_head= 7;
end
