`timescale 1ns / 1ps

module fifo( input clk, input reset, input rd, input wr, input [0:8-1]w_data, output [0:8-1]r_data, output empty, output full);

integer head=0,tail=0;
reg [7:0] cque[0:7];
reg [7:0]b;
reg f,e;
initial begin
    e = 1;
    f = 0;
    head = 0;
    tail = 0;
    b="";
end

always @(posedge clk) 
begin

    if ( reset == 1 )begin
        cque[0]=""; 
        cque[1]=""; 
        cque[2]=""; 
        cque[3]=""; 
        cque[4]=""; 
        cque[5]=""; 
        cque[6]=""; 
        cque[7]=""; 
    end 
    if ( reset == 0 )begin
        if( head == tail && cque[(head+7)%8] != "")begin
            f = 1;
        end
        if( head != tail || cque[(head+7)%8] == "")begin
            f=0;
        end
        if( tail != head)begin
            e = 0;
        end
        if( tail == head && cque[(head+7)%8] == "")begin
            e = 1;
        end
        if( rd == 1 && e == 0)begin 
            b = cque[tail];
            cque[tail]="";
            tail = ( tail + 1 ) % 8;
        end
        if( wr == 1 && rd == 0 && f != 1 )begin 
            cque[head] = w_data;
            head = (head + 1) % 8; 
        end
        if(rd == 0 || e == 1 )begin
            b="";
        end
    end 
end
assign full = f;
assign empty = e;
assign r_data = b;
endmodule
