module fifo(clk,reset,rd,wr,w_data,r_data,empty,full);
input wire clk,reset,rd,wr;
input[8-1:0] w_data;
output reg [8-1:0] r_data;
output reg empty,full;
reg [8-1:0] mem[9-1:0];
integer head,tail;
integer i;


always @( posedge clk) begin

if(reset==1)begin
//count=0;
head=0;
tail=0;
for(i=0;i<9;i=i+1)
mem[i]="";
end

else begin

if(head==tail) begin
empty=1'b1;
r_data="";
end
else
empty = 1'b0;

if(head==(tail+1)%9 )
full=1'b1;
else
full = 1'b0;

if(rd && !empty && !wr) begin //for read
head=(head+1)%9;
r_data=mem[head];
end

if(wr && !full && !rd ) begin// for write
tail= (tail+1)%9;
mem[tail]=w_data;
end

end
end
endmodule
