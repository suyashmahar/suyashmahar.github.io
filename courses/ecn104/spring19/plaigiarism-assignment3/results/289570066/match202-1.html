module fifo(clk,reset,rd,wr,w_data,r_data,empty,full);
    input clk;
    input reset;
    input rd;
    input wr;
    input [0:7] w_data ;
  
    reg [7:0] store [0:7];

    output reg empty ;
    output reg full ;
    output reg [0:7] r_data ;
     
    integer f=-1, r=-1;
    
    initial begin
        full = 0;
        empty = 1;
        store[0] = " ";
        store[1] = " ";
        store[2] = " ";
        store[3] = " ";
        store[4] = " ";
        store[5] = " ";
        store[6] = " ";
        store[7] = " ";
    end
    
    always @(posedge clk) 
    begin
    
        if(wr==1&&f<0) begin
            f = 0;
            r = 0;
            store[f] = w_data;
            empty = 0;
            end
            
        else if(wr==1&&f>=0&&f<7) begin
            f = f+1;
            store[f] = w_data;
        end
            
        else if(wr==1&&f==7) begin
            empty = 0;
            full = 1;
        end
       
        else if(rd==1&r==-1) begin
            empty = 1;
            full = 0;     
        end
        
        else if(rd==1&r>=0&&r<7) begin
            r_data = store[r];     
             r = r+1;
        end
        
        else if(rd==1&r==7) begin
            empty = 0 ;
            full = 1;
            r_data = store[r];     
        end
        
        else if(reset==1) begin
            store[0] = " ";
            store[1] = " ";
            store[2] = " ";
            store[3] = " ";
            store[4] = " ";
            store[5] = " ";
            store[6] = " ";
            store[7] = " ";
            f = -1;
            r = -1;
            empty = 1;
            full = 0;
        end
       
    end

endmodule
