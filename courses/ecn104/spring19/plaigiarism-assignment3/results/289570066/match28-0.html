module fifo(clk,reset,wr,w_data[7:0],rd,r_data[7:0],empty,full);
      input wire clk;
      input reset;
      input rd;
      input wr;
      input [7:0] w_data;
      output reg[7:0] r_data;
      output reg empty;
      output reg full;
      integer im;
      integer jk;
      integer line[7:0];
       always @(posedge clk)
       begin
       if(reset==1)
       begin
       line[0]=0;
       line[1]=0;
       line[2]=0;
       line[3]=0;


       line[4]=0;
       line[5]=0;
       line[6]=0;
       im=-1;
       jk=-1;
       empty=1;
       full=0;
       end
       else
       begin
       if((wr==1)&&(full==0))


       begin
       empty=0;
       if((im==-1)&&(jk==-1))
       begin
       line[0]=w_data;
       im=0;
       jk=0;
       end
       else if(jk==7)
       begin
       line[jk]=w_data;
       end
       else
       begin
       jk=jk+1;
       line[jk]=w_data;
       end
       if((im==0)&&(jk==7))
       full =1;
       else if(im==(jk+1))
       full=1;
       end
       if((rd==1)&&(empty==0))
       begin
       r_data=line[im];
       line[im]=0;
       if(im==jk)
       begin
       im=-1;
       jk=-1;
       empty=1;
       end
       else
       begin
       im=(im+1)*8;
       end
       end
       end
       end
       
endmodule
