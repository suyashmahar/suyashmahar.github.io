module fifo(clk,reset,rd,wr,w_data[7:0],r_data[7:0],empty,full);
input wire clk;
input reset;
input rd;
input wr;
input [7:0] w_data;
output reg [7:0] r_data;
output reg empty;
output reg full;
integer front;
integer rear;
integer queue[7:0];

always @(posedge clk)
   begin
        if(reset==1)
            begin
                 queue[0]=0;
                 queue[1]=0;
                 queue[2]=0;
                 queue[3]=0;
                 queue[4]=0;
                 queue[5]=0;
                 queue[6]=0;
                 queue[7]=0;
                 front=-1;
                 rear=-1;
                 empty=1;
                 full=0;
             end
             else
                 begin
        if ((wr==1)&&(full==0))
             begin
                 empty=0;
                 if((front==-1) && (full==0))
                     begin
                         queue[0]=w_data;
                         front=0;
                         rear=0;
                     end
                 else if(rear==7)
                     begin
                         queue[rear] =w_data;
                     end
                 else
                     begin
                         rear=rear+1;
                         queue[rear]=w_data;
                     end
                 
                 if((front==0) && (rear==7))
                       full=1;
                 else if (front == (rear+1))
                       full=1;
                       
                       
              end
              if((rd==1) && (empty==0))
                  begin
                      r_data=queue[front];
                      queue[front]=0;
                      if (front==rear)
                           begin
                               front=-1;
                               rear=-1;
                               empty=1;
                           end
               else
                  begin
                      front=(front+1)%8;
                  end
              end
              end
              end
endmodule
