module fifo(clk, reset, rd, wr, w_data, r_data, empty, full);
input clk, reset, rd, wr;
input [7:0]w_data;
output reg [7:0]r_data;
output reg empty, full;

reg [7:0]que[7:0];
integer count=0;
integer wptr=0, rptr=0;

initial begin
assign empty=1'b1;
assign full=1'b0;
end

always @(posedge clk) begin

if(reset) begin
wptr=0;
rptr=0;
count=0;
assign empty=1'b1;
assign full=1'b0;
end

else begin 

if(wr&&count<9) begin
que[wptr]=w_data;
wptr=(wptr+1)%8;
count=count+1;
assign empty=1'b0;
end

if(rd&&count!=0)  begin
assign r_data=que[rptr];
rptr=(rptr+1)%8;
count=count-1;
end

if(count==0) begin
assign empty=1'b1;
end
if(count==8) begin
assign full=1'b1;
end
else begin
assign full=1'b0;
end


end
end
endmodule
